javascript $(document).ready(function() { // Set up the click event handler for the search button $('#searchButton').click(function() { // Get the search query from the textarea var searchQuery = $('#searchQuery').val(); // Call the search API using JSONP $.ajax({ url: 'https://www.youtube.com/results', data: { search_query: searchQuery }, dataType: 'html', method: 'GET', success: function(data) { // Extract the video IDs from the search results using regex var videoIds = []; var regex = /"videoId":"([^"]+)"/g; var match; while (match = regex.exec(data)) { videoIds.push(match[1]); } // Display the video IDs in the div $('#videoIds').html(videoIds.join('
')); }, error: function(xhr, status, error) { console.log('Error:', error); } }); }); });
